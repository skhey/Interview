# Asynchronous Patterns Beyond Promises – Complete Guide

Node.js provides powerful asynchronous primitives beyond basic Promises and `async/await`. These patterns give you fine-grained control over complex async flows, resource cleanup, cancellation, and execution tracking.

---

## Async Iterators (`for await...of`)

### The Problem with Traditional Async Loops

Processing async data sequentially often leads to nested callbacks or complex Promise chains:

```javascript
// ❌ Cumbersome with regular Promises
async function processItems(items) {
  for (const item of items) {
    await processItem(item);  // Sequential, but blocks the loop
  }
}

// ❌ Parallel is easy but loses order control
await Promise.all(items.map(processItem));
```

**Async iterators** solve this by providing a **pull-based model** where you consume values as they become available, with built-in backpressure handling.

### Consuming Async Iterators

```javascript
// Consuming an async iterable (e.g., readable stream)
async function consumeStream(stream) {
  for await (const chunk of stream) {
    console.log(`Received ${chunk.length} bytes`);
    await processChunk(chunk);  // Process sequentially with backpressure
  }
  console.log('Stream ended');
}

// Real example: Reading lines from a large file
const fs = require('node:fs');
const readline = require('node:readline');

async function processLargeFile(filepath) {
  const fileStream = fs.createReadStream(filepath);
  const rl = readline.createInterface({ input: fileStream });
  
  let lineCount = 0;
  for await (const line of rl) {
    lineCount++;
    if (lineCount % 10000 === 0) {
      console.log(`Processed ${lineCount} lines`);
    }
  }
  console.log(`Total: ${lineCount} lines`);
}
```

### Creating Custom Async Iterators

**Using `Symbol.asyncIterator`:**

```javascript
// Custom async iterable that yields values with a delay
const delayedRange = {
  from: 1,
  to: 5,
  delay: 1000,
  
  [Symbol.asyncIterator]() {
    return {
      current: this.from,
      last: this.to,
      delay: this.delay,
      
      async next() {
        await new Promise(resolve => setTimeout(resolve, this.delay));
        
        if (this.current <= this.last) {
          return { done: false, value: this.current++ };
        } else {
          return { done: true };
        }
      }
    };
  }
};

// Usage
for await (const num of delayedRange) {
  console.log(num); // 1, 2, 3, 4, 5 (each after 1 second)
}
```

**Using Async Generator Functions (`async function*`):**

```javascript
// Async generator - simplest way to create async iterables
async function* paginatedAPI(baseUrl, limit = 100) {
  let page = 1;
  let hasMore = true;
  
  while (hasMore) {
    const response = await fetch(`${baseUrl}?page=${page}&limit=${limit}`);
    const data = await response.json();
    
    // Yield each item one by one
    for (const item of data.items) {
      yield item;
    }
    
    hasMore = data.hasMore;
    page++;
  }
}

// Usage
for await (const user of paginatedAPI('https://api.example.com/users')) {
  console.log(user.name);
  // Items are fetched page-by-page, but consumed one-by-one
}
```

### Real-World: Database Cursor Streaming

```javascript
const { MongoClient } = require('mongodb');

async function* streamLargeCollection(collectionName, query = {}) {
  const client = new MongoClient(process.env.MONGO_URL);
  await client.connect();
  
  const collection = client.db().collection(collectionName);
  const cursor = collection.find(query).stream();
  
  try {
    for await (const doc of cursor) {
      yield doc;
    }
  } finally {
    await client.close();  // Ensures cleanup even if consumer breaks early
  }
}

// Usage with memory-efficient processing
async function processAllUsers() {
  for await (const user of streamLargeCollection('users', { active: true })) {
    await sendWelcomeEmail(user);
    // Only one user document in memory at a time
  }
}
```

### Async Iterator Helpers (ES2025 / Stage 3)

```javascript
// Map over async iterable
async function* map(asyncIterable, fn) {
  for await (const item of asyncIterable) {
    yield fn(item);
  }
}

// Filter async iterable
async function* filter(asyncIterable, predicate) {
  for await (const item of asyncIterable) {
    if (await predicate(item)) {
      yield item;
    }
  }
}

// Take first N items
async function* take(asyncIterable, n) {
  let count = 0;
  for await (const item of asyncIterable) {
    if (count++ >= n) break;
    yield item;
  }
}

// Usage
const userStream = paginatedAPI('https://api.example.com/users');
const activeUsers = filter(userStream, user => user.active);
const firstTen = take(activeUsers, 10);

for await (const user of firstTen) {
  console.log(user);
}
```

## Event Emitters

The `EventEmitter` class is Node.js's implementation of the **Observer pattern**. It's the foundation of many core modules (`http`, `stream`, `process`, `net`).

### Basic EventEmitter Usage

```javascript
const EventEmitter = require('node:events');

class OrderProcessor extends EventEmitter {
  constructor() {
    super();
    this.orders = new Map();
  }
  
  createOrder(orderData) {
    const orderId = `ORD-${Date.now()}`;
    this.orders.set(orderId, { ...orderData, status: 'pending' });
    
    // Emit events
    this.emit('order:created', { orderId, data: orderData });
    
    // Simulate async processing
    setTimeout(() => {
      this.orders.get(orderId).status = 'confirmed';
      this.emit('order:confirmed', { orderId });
    }, 1000);
    
    return orderId;
  }
  
  cancelOrder(orderId) {
    const order = this.orders.get(orderId);
    if (!order) {
      this.emit('error', new Error(`Order ${orderId} not found`));
      return false;
    }
    
    this.orders.delete(orderId);
    this.emit('order:cancelled', { orderId });
    return true;
  }
}

// Usage
const processor = new OrderProcessor();

processor.on('order:created', ({ orderId, data }) => {
  console.log(`Order ${orderId} created for ${data.customer}`);
  sendConfirmationEmail(data.customer, orderId);
});

processor.on('order:confirmed', ({ orderId }) => {
  console.log(`Order ${orderId} confirmed`);
  notifyWarehouse(orderId);
});

processor.on('order:cancelled', ({ orderId }) => {
  console.log(`Order ${orderId} cancelled`);
});

processor.on('error', (err) => {
  console.error('Order processing error:', err.message);
});

// Create orders
processor.createOrder({ customer: 'alice@example.com', items: ['item1'] });
```

### Real-World: Custom EventEmitter with Error Handling

```javascript
const EventEmitter = require('node:events');

class DatabaseConnection extends EventEmitter {
  constructor(config) {
    super();
    this.config = config;
    this.connected = false;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
  }
  
  async connect() {
    try {
      // Simulate connection
      await new Promise(resolve => setTimeout(resolve, 100));
      this.connected = true;
      this.reconnectAttempts = 0;
      this.emit('connected');
      
      // Start heartbeat
      this.startHeartbeat();
    } catch (err) {
      this.emit('error', err);
      this.attemptReconnect();
    }
  }
  
  startHeartbeat() {
    this.heartbeatInterval = setInterval(() => {
      if (this.connected) {
        this.emit('heartbeat', { timestamp: Date.now() });
      }
    }, 30000);
  }
  
  attemptReconnect() {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      this.emit('error', new Error('Max reconnect attempts reached'));
      return;
    }
    
    this.reconnectAttempts++;
    this.emit('reconnecting', { attempt: this.reconnectAttempts });
    
    setTimeout(() => this.connect(), 1000 * this.reconnectAttempts);
  }
  
  async disconnect() {
    clearInterval(this.heartbeatInterval);
    this.connected = false;
    this.emit('disconnected');
  }
}

// Usage with proper error handling
const db = new DatabaseConnection({ host: 'localhost' });

db.on('connected', () => console.log('Database connected'));
db.on('disconnected', () => console.log('Database disconnected'));
db.on('reconnecting', ({ attempt }) => console.log(`Reconnecting... attempt ${attempt}`));
db.on('heartbeat', ({ timestamp }) => console.log(`Heartbeat at ${new Date(timestamp)}`));
db.on('error', (err) => {
  console.error('Database error:', err.message);
  // Notify monitoring system
});

db.connect();

// Graceful shutdown
process.on('SIGTERM', async () => {
  await db.disconnect();
  process.exit(0);
});
```

## Callbackify (`util.callbackify`)

### The Problem: Integrating Promises with Legacy Callback APIs

Many older Node.js APIs and libraries expect **error-first callbacks** (also called "nodebacks"):

```javascript
// Legacy callback API
function legacyFunction(arg, callback) {
  setTimeout(() => {
    if (arg === 'error') {
      callback(new Error('Something went wrong'));
    } else {
      callback(null, `Result: ${arg}`);
    }
  }, 100);
}
```

When you have a modern Promise-based function but need to integrate with a callback API, `util.callbackify` converts it.

### Basic Usage of `callbackify`

```javascript
const util = require('node:util');

// Modern async function
async function fetchUserData(userId) {
  const response = await fetch(`https://api.example.com/users/${userId}`);
  if (!response.ok) throw new Error(`HTTP ${response.status}`);
  return response.json();
}

// Convert to error-first callback style
const fetchUserDataCallback = util.callbackify(fetchUserData);

// Use with legacy callback API
fetchUserDataCallback(123, (err, data) => {
  if (err) {
    console.error('Failed to fetch user:', err.message);
    return;
  }
  console.log('User data:', data);
});
```

### Advanced: Multiple Arguments and Error Handling

```javascript
const util = require('node:util');

// Async function that returns multiple values (as object or array)
async function getMultipleResults() {
  const user = await fetch('/user').then(r => r.json());
  const posts = await fetch('/posts').then(r => r.json());
  return { user, posts };
}

const getMultipleCallback = util.callbackify(getMultipleResults);

getMultipleCallback((err, result) => {
  if (err) return console.error(err);
  // result is { user: {...}, posts: [...] }
  console.log(result.user, result.posts);
});

// IMPORTANT: The callback only receives the FIRST value if multiple are resolved.
// Async functions always return a single value.
```

### Real-World: Wrapping a Promise-Based Database Client for Legacy Code

```javascript
const util = require('node:util');
const { MongoClient } = require('mongodb');

class LegacyMongoWrapper {
  constructor(url) {
    this.client = new MongoClient(url);
    this.db = null;
    
    // Convert promise methods to callbacks
    this.connect = util.callbackify(this._connect.bind(this));
    this.find = util.callbackify(this._find.bind(this));
    this.insert = util.callbackify(this._insert.bind(this));
  }
  
  async _connect() {
    await this.client.connect();
    this.db = this.client.db();
    return { connected: true };
  }
  
  async _find(collection, query) {
    if (!this.db) throw new Error('Not connected');
    return this.db.collection(collection).find(query).toArray();
  }
  
  async _insert(collection, doc) {
    if (!this.db) throw new Error('Not connected');
    const result = await this.db.collection(collection).insertOne(doc);
    return { insertedId: result.insertedId };
  }
}

// Usage in legacy codebase
const db = new LegacyMongoWrapper('mongodb://localhost:27017');

db.connect((err, result) => {
  if (err) throw err;
  console.log('Connected:', result.connected);
  
  db.insert('users', { name: 'Alice' }, (err, result) => {
    if (err) throw err;
    console.log('Inserted:', result.insertedId);
    
    db.find('users', {}, (err, users) => {
      if (err) throw err;
      console.log('Users:', users);
    });
  });
});
```

### Custom Implementation (Without `util.callbackify`)

```javascript
function promiseToCallback(promiseFn) {
  return function(...args) {
    const callback = args.pop(); // Last argument is callback
    
    promiseFn(...args)
      .then(result => callback(null, result))
      .catch(err => callback(err));
  };
}
```

## AbortController

`AbortController` provides a **standardized way to cancel asynchronous operations**. It's available globally in Node.js (since v15) and browsers.

### Basic AbortController Usage

```javascript
// Create controller
const controller = new AbortController();
const signal = controller.signal;

// Start an async operation with the signal
fetch('https://api.example.com/slow-endpoint', { signal })
  .then(response => response.json())
  .then(data => console.log(data))
  .catch(err => {
    if (err.name === 'AbortError') {
      console.log('Request was cancelled');
    } else {
      console.error('Request failed:', err);
    }
  });

// Cancel after 2 seconds
setTimeout(() => {
  controller.abort();
  console.log('Request aborted');
}, 2000);
```

### Cancelling Multiple Operations with One Signal

```javascript
const controller = new AbortController();
const { signal } = controller;

// Cancel multiple operations at once
Promise.all([
  fetch('/api/users', { signal }),
  fetch('/api/posts', { signal }),
  fetch('/api/comments', { signal })
]).catch(err => {
  if (err.name === 'AbortError') {
    console.log('All requests cancelled');
  }
});

// Cancel everything
setTimeout(() => controller.abort(), 5000);
```

### AbortController with Timers

```javascript
// Cancellable timeout
function cancellableTimeout(ms, signal) {
  return new Promise((resolve, reject) => {
    const timeout = setTimeout(resolve, ms);
    
    signal.addEventListener('abort', () => {
      clearTimeout(timeout);
      reject(new DOMException('Timeout aborted', 'AbortError'));
    });
  });
}

// Usage
const controller = new AbortController();

cancellableTimeout(5000, controller.signal)
  .then(() => console.log('5 seconds passed'))
  .catch(err => console.log('Cancelled:', err.message));

// Cancel early
setTimeout(() => controller.abort(), 1000);
```

### AbortController with Event Listeners

```javascript
function createAbortableEventListener(target, eventName, signal) {
  return new Promise((resolve, reject) => {
    const handler = (event) => {
      target.removeEventListener(eventName, handler);
      resolve(event);
    };
    
    target.addEventListener(eventName, handler);
    
    signal.addEventListener('abort', () => {
      target.removeEventListener(eventName, handler);
      reject(new DOMException('Event listener aborted', 'AbortError'));
    });
  });
}

// Usage
const controller = new AbortController();

createAbortableEventListener(process, 'SIGINT', controller.signal)
  .then(() => console.log('Received SIGINT'))
  .catch(() => console.log('Aborted before SIGINT'));

// Cancel after 10 seconds
setTimeout(() => controller.abort(), 10000);
```

### Real-World: AbortController in Express Route

```javascript
const express = require('express');
const app = express();

app.get('/search', async (req, res) => {
  const controller = new AbortController();
  const { signal } = controller;
  
  // Set timeout for the entire operation
  const timeout = setTimeout(() => controller.abort(), 10000);
  
  // Handle client disconnect
  req.on('close', () => {
    console.log('Client disconnected, aborting...');
    controller.abort();
  });
  
  try {
    const query = req.query.q;
    
    // Multiple parallel requests with shared abort signal
    const [products, categories, suggestions] = await Promise.all([
      fetch(`https://api.example.com/products?q=${query}`, { signal }),
      fetch(`https://api.example.com/categories?q=${query}`, { signal }),
      fetch(`https://api.example.com/suggestions?q=${query}`, { signal })
    ].map(p => p.then(r => r.json())));
    
    clearTimeout(timeout);
    res.json({ products, categories, suggestions });
    
  } catch (err) {
    clearTimeout(timeout);
    
    if (err.name === 'AbortError') {
      res.status(408).json({ error: 'Request timeout or client disconnected' });
    } else {
      res.status(500).json({ error: err.message });
    }
  }
});
```

### Chaining AbortControllers with Timeout

```javascript
function withTimeout(signal, ms) {
  const controller = new AbortController();
  
  const timeout = setTimeout(() => controller.abort(), ms);
  
  // Forward abort from parent signal
  signal.addEventListener('abort', () => {
    clearTimeout(timeout);
    controller.abort();
  });
  
  // Clean up timeout on abort
  controller.signal.addEventListener('abort', () => clearTimeout(timeout));
  
  return controller.signal;
}

// Usage
const mainController = new AbortController();
const timeoutSignal = withTimeout(mainController.signal, 5000);

fetch('https://api.example.com/slow', { signal: timeoutSignal })
  .catch(err => {
    if (err.name === 'AbortError') {
      console.log('Either timeout or manual abort');
    }
  });

// Can be aborted manually or times out automatically
setTimeout(() => mainController.abort(), 2000);
```

### AbortController with Streams

```javascript
const fs = require('node:fs');
const { pipeline } = require('node:stream/promises');

async function copyFileWithCancellation(src, dest, signal) {
  const controller = new AbortController();
  
  // Forward external signal
  signal.addEventListener('abort', () => controller.abort());
  
  try {
    await pipeline(
      fs.createReadStream(src),
      new Transform({
        transform(chunk, encoding, callback) {
          // Simulate slow processing
          setTimeout(() => callback(null, chunk), 100);
        },
        signal: controller.signal  // Node.js 18+ supports signal in Transform
      }),
      fs.createWriteStream(dest),
      { signal: controller.signal }
    );
  } catch (err) {
    if (err.name === 'AbortError') {
      console.log('File copy cancelled');
      // Clean up partial file
      fs.unlink(dest, () => {});
    }
    throw err;
  }
}
```

## async_hooks (Advanced Execution Context Tracking)

`async_hooks` provides a low-level API to **track asynchronous resources** throughout their lifecycle. It's the foundation for APM tools, distributed tracing, and debugging.

> ⚠️ **Warning:** `async_hooks` is a **powerful but complex** API. Use with caution in production; it has performance implications and can cause memory leaks if not handled correctly.

### Core Concepts

| Concept | Description |
|---------|-------------|
| **Async Resource** | Any object with an associated callback (e.g., `Promise`, `Timeout`, `TCP` socket) |
| **Execution Context** | The current async context (which async operation is running) |
| **Async ID** | Unique identifier for each async resource |
| **Trigger ID** | The async ID that caused this resource to be created |

### Basic async_hooks Usage

```javascript
const async_hooks = require('node:async_hooks');
const fs = require('node:fs');

// Create hook to track async operations
const hook = async_hooks.createHook({
  init(asyncId, type, triggerAsyncId, resource) {
    const eid = async_hooks.executionAsyncId();
    fs.writeSync(1, `${type}(${asyncId}): trigger: ${triggerAsyncId} exec: ${eid}\n`);
  },
  before(asyncId) {
    fs.writeSync(1, `before: ${asyncId}\n`);
  },
  after(asyncId) {
    fs.writeSync(1, `after: ${asyncId}\n`);
  },
  destroy(asyncId) {
    fs.writeSync(1, `destroy: ${asyncId}\n`);
  },
  promiseResolve(asyncId) {
    fs.writeSync(1, `promiseResolve: ${asyncId}\n`);
  }
});

hook.enable();

// Trigger some async operations
setTimeout(() => {
  console.log('Timeout fired');
  Promise.resolve().then(() => console.log('Promise resolved'));
}, 100);

// Later...
// hook.disable();
```

**Sample Output:**
```

before: 2
Timeout fired
after: 2
PROMISE(3): trigger: 2 exec: 2
promiseResolve: 3
Promise resolved
destroy: 2
```

### Creating AsyncLocalStorage for Request Context

`AsyncLocalStorage` (built on `async_hooks`) is the **production-friendly way** to maintain context across async operations:

```javascript
const { AsyncLocalStorage } = require('node:async_hooks');

const als = new AsyncLocalStorage();

// Express middleware for request tracing
app.use((req, res, next) => {
  // Create a unique request ID
  const requestId = `req-${Date.now()}-${Math.random().toString(36)}`;
  
  // Store context for this request
  als.run({ requestId, userId: req.user?.id, path: req.path }, () => {
    // All async operations inside this callback share the context
    next();
  });
});

// Access context anywhere in the call chain
function logWithContext(message) {
  const store = als.getStore();
  const requestId = store?.requestId || 'unknown';
  console.log(`[${requestId}] ${message}`);
}

app.get('/users/:id', async (req, res) => {
  logWithContext(`Fetching user ${req.params.id}`);
  
  const user = await fetchUser(req.params.id);
  logWithContext(`User found: ${user.name}`);
  
  res.json(user);
});
```

### Real-World: Distributed Tracing with AsyncLocalStorage

```javascript
const { AsyncLocalStorage } = require('node:async_hooks');
const als = new AsyncLocalStorage();

class Tracer {
  static startTrace(name, metadata = {}) {
    const store = als.getStore() || {};
    const traceId = store.traceId || this.generateTraceId();
    const spanId = this.generateSpanId();
    
    const span = {
      traceId,
      spanId,
      parentSpanId: store.spanId,
      name,
      startTime: Date.now(),
      metadata
    };
    
    return als.run({ ...store, traceId, spanId, span }, () => span);
  }
  
  static endTrace(span, metadata = {}) {
    span.endTime = Date.now();
    span.duration = span.endTime - span.startTime;
    span.metadata = { ...span.metadata, ...metadata };
    
    // Send to tracing backend (e.g., Jaeger, Zipkin)
    this.sendToCollector(span);
  }
  
  static getCurrentSpan() {
    return als.getStore()?.span;
  }
  
  static addMetadata(key, value) {
    const store = als.getStore();
    if (store?.span) {
      store.span.metadata[key] = value;
    }
  }
  
  static generateTraceId() {
    return require('crypto').randomBytes(16).toString('hex');
  }
  
  static generateSpanId() {
    return require('crypto').randomBytes(8).toString('hex');
  }
  
  static sendToCollector(span) {
    // In production: send to tracing backend
    console.log('Span:', JSON.stringify(span, null, 2));
  }
}

// Express middleware
app.use((req, res, next) => {
  const span = Tracer.startTrace(`${req.method} ${req.path}`, {
    type: 'http',
    method: req.method,
    path: req.path,
    remoteIp: req.ip
  });
  
  // Add response metadata
  res.on('finish', () => {
    Tracer.endTrace(span, {
      statusCode: res.statusCode,
      responseTime: Date.now() - span.startTime
    });
  });
  
  next();
});

// Usage in business logic
```

### Real-World: Connection Pool Leak Detection

```javascript
const async_hooks = require('node:async_hooks');
const fs = require('node:fs');

class ResourceTracker {
  constructor() {
    this.resources = new Map();
    this.hook = async_hooks.createHook({
      init: (asyncId, type, triggerAsyncId) => {
        if (type === 'TCPSERVERWRAP' || type === 'TCPWRAP') {
          const stack = new Error().stack;
          this.resources.set(asyncId, {
            type,
            triggerAsyncId,
            createdAt: Date.now(),
            stack
          });
        }
      },
      destroy: (asyncId) => {
        this.resources.delete(asyncId);
      }
    });
  }
  
  start() {
    this.hook.enable();
  }
  
  stop() {
    this.hook.disable();
  }
  
  report() {
    const now = Date.now();
    const leaks = [];
    
    for (const [id, info] of this.resources) {
      const age = now - info.createdAt;
      if (age > 30000) { // Older than 30 seconds
        leaks.push({ id, age, type: info.type, stack: info.stack });
      }
    }
    
    if (leaks.length > 0) {
      fs.writeFileSync('resource-leaks.json', JSON.stringify(leaks, null, 2));
      console.warn(`Found ${leaks.length} potential resource leaks`);
    }
    
    return leaks;
  }
}

// Usage
const tracker = new ResourceTracker();
tracker.start();

// Check for leaks every minute
setInterval(() => tracker.report(), 60000);
```

## Combining Patterns: Complete Real-World Example

### File Upload Service with All Patterns

```javascript
const express = require('express');
const multer = require('multer');
const { EventEmitter, once, on } = require('node:events');
const { AsyncLocalStorage } = require('node:async_hooks');
const { pipeline } = require('node:stream/promises');
const fs = require('node:fs');
const util = require('node:util');

// AsyncLocalStorage for request context
const als = new AsyncLocalStorage();

// Custom EventEmitter for upload events
class UploadProcessor extends EventEmitter {
  constructor() {
    super();
    this.activeUploads = new Map();
  }
  
  async *processUpload(filePath) {
    const fileStream = fs.createReadStream(filePath);
    const rl = readline.createInterface({ input: fileStream });
    
    let lineNumber = 0;
    for await (const line of rl) {
      lineNumber++;
      
      // Yield progress events
      yield { type: 'progress', line: lineNumber };
      
      if (lineNumber % 1000 === 0) {
        this.emit('chunk:processed', { filePath, lineNumber });
      }
    }
    
    yield { type: 'complete', totalLines: lineNumber };
    this.emit('upload:complete', { filePath, totalLines: lineNumber });
  }
}

// Abortable upload handler
async function handleUpload(filePath, signal) {
  const controller = new AbortController();
  const combinedSignal = controller.signal;
  
  signal.addEventListener('abort', () => controller.abort());
  
  const processor = new UploadProcessor();
  
  try {
    for await (const event of on(processor.processUpload(filePath), 'data')) {
      if (combinedSignal.aborted) {
        throw new DOMException('Upload cancelled', 'AbortError');
      }
      
      const store = als.getStore();
      console.log(`[${store?.requestId}] Upload progress:`, event);
    }
  } catch (err) {
    if (err.name === 'AbortError') {
      // Clean up partial upload
      await fs.promises.unlink(filePath);
    }
    throw err;
  }
}

// Convert to callback for legacy integration
const handleUploadCallback = util.callbackify(handleUpload);

// Express app
const app = express();
const upload = multer({ dest: 'uploads/' });

app.use((req, res, next) => {
  const requestId = req.headers['x-request-id'] || `req-${Date.now()}`;
  
  als.run({ requestId, path: req.path }, () => {
    next();
  });
});

app.post('/upload', upload.single('file'), async (req, res) => {
  const controller = new AbortController();
  
  // Timeout after 5 minutes
  const timeout = setTimeout(() => controller.abort(), 5 * 60 * 1000);
  
  // Client disconnect cancels upload
  req.on('close', () => controller.abort());
  
  try {
    await handleUpload(req.file.path, controller.signal);
    clearTimeout(timeout);
    res.json({ status: 'complete', file: req.file.filename });
  } catch (err) {
    clearTimeout(timeout);
    
    if (err.name === 'AbortError') {
      res.status(499).json({ error: 'Upload cancelled' });
    } else {
      res.status(500).json({ error: err.message });
    }
  }
});

app.listen(3000);
```

## Summary Table: When to Use Each Pattern

| Pattern | Best For | Example Use Cases |
|---------|----------|-------------------|
| **Async Iterators** | Sequential async data processing with backpressure | Reading large files line-by-line, paginated API consumption |
| **Event Emitters** | Decoupled event-driven architectures | Real-time notifications, progress tracking, custom hooks |
| **Callbackify** | Integrating modern async code with legacy callback APIs | Wrapping Promise libraries for use with Express middleware |
| **AbortController** | Cancellable async operations | Timeouts, user-initiated cancellation, cleanup on disconnect |
| **AsyncLocalStorage** | Maintaining context across async boundaries | Request tracing, logging correlation, distributed tracing |
| **async_hooks** | Low-level resource tracking (use sparingly) | APM tools, debugging, leak detection |

**Key Takeaways:**
1. **Use AsyncLocalStorage, not raw async_hooks** — it's optimized and production-safe
2. **Always pass AbortSignal** to cancellable operations — it's the standard
3. **Async iterators + `for await...of`** replace complex stream/callback patterns
4. **EventEmitter + `events.on()`** gives you async iterable event consumption
5. **`util.callbackify`** is a bridge — prefer Promises for new code
6. **Combine patterns** for robust production systems (e.g., AbortController + AsyncLocalStorage + EventEmitter)
