# Node.js Security – Enterprise Guide

Security in Node.js applications is non‑negotiable, especially in enterprise environments handling sensitive user data, financial transactions, or compliance requirements. This guide covers the essential security domains every production Node.js application must address.

---

## 1. Access Control

Access control determines **who can do what** in your application. It goes beyond simple authentication to enforce fine‑grained permissions.

### Role‑Based Access Control (RBAC)

Users are assigned **roles**, and roles are granted **permissions**.

```
┌─────────────┐     ┌─────────────┐     ┌─────────────────┐
│    User     │────▶│    Role     │────▶│   Permission    │
│   Alice     │     │   Admin     │     │  users:delete   │
└─────────────┘     └─────────────┘     └─────────────────┘
```

**Implementation:**

```javascript
// permissions.js
const permissions = {
  admin: [
    'users:read', 'users:create', 'users:update', 'users:delete',
    'reports:view', 'settings:edit'
  ],
  manager: [
    'users:read', 'users:create', 'users:update',
    'reports:view'
  ],
  user: [
    'users:read:self',
    'profile:edit:self'
  ]
};

function hasPermission(userRole, requiredPermission) {
  return permissions[userRole]?.includes(requiredPermission) || false;
}

// Middleware
function requirePermission(permission) {
  return (req, res, next) => {
    const userRole = req.user?.role;
    if (!userRole) {
      return res.status(401).json({ error: 'Unauthorized' });
    }
    if (!hasPermission(userRole, permission)) {
      return res.status(403).json({ error: 'Forbidden' });
    }
    next();
  };
}

// Usage
app.delete('/users/:id', 
  authenticate, 
  requirePermission('users:delete'), 
  deleteUserHandler
);
```

### Attribute‑Based Access Control (ABAC)

ABAC evaluates **attributes** of the user, resource, and environment to make access decisions.

```javascript
// ABAC policy engine
class ABACEngine {
  constructor() {
    this.policies = [];
  }
  
  addPolicy(policy) {
    this.policies.push(policy);
  }
  
  evaluate(user, resource, action, environment = {}) {
    for (const policy of this.policies) {
      const matches = policy.condition({ user, resource, action, environment });
      if (matches) {
        return policy.effect === 'allow';
      }
    }
    return false; // Default deny
  }
}

// Define policies
const abac = new ABACEngine();

abac.addPolicy({
  name: 'user-can-edit-own-document',
  effect: 'allow',
  condition: ({ user, resource, action }) => {
    return action === 'edit' && 
           resource.type === 'document' && 
           resource.ownerId === user.id;
  }
});

abac.addPolicy({
  name: 'admin-can-do-anything',
  effect: 'allow',
  condition: ({ user }) => user.role === 'admin'
});

abac.addPolicy({
  name: 'ip-restriction',
  effect: 'allow',
  condition: ({ environment }) => {
    const allowedIPs = ['10.0.0.0/8', '192.168.0.0/16'];
    return allowedIPs.some(cidr => ipInCIDR(environment.ip, cidr));
  }
});

// Middleware
function abacMiddleware(action) {
  return async (req, res, next) => {
    const user = req.user;
    const resource = await getResource(req.params.id);
    const environment = { ip: req.ip, time: new Date() };
    
    if (abac.evaluate(user, resource, action, environment)) {
      next();
    } else {
      res.status(403).json({ error: 'Forbidden' });
    }
  };
}
```

### Hierarchical / Scoped Access (Multi‑Tenancy)

```javascript
// Tenant‑aware access control
function requireTenantAccess(paramName = 'tenantId') {
  return (req, res, next) => {
```

## 2. JWT + OAuth2

### JSON Web Tokens (JWT)

JWTs are **stateless tokens** containing signed claims about the user. Best for distributed systems and microservices.

**JWT Structure:**
```
eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJzdWIiOiIxMjM0NTY3ODkwIiwibmFtZSI6IkpvaG4gRG9lIiwiaWF0IjoxNTE2MjM5MDIyfQ.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c

Header:   { "alg": "HS256", "typ": "JWT" }
Payload:  { "sub": "1234567890", "name": "John Doe", "iat": 1516239022 }
Signature: HMACSHA256(base64UrlEncode(header) + "." + base64UrlEncode(payload), secret)
```

**Secure JWT Implementation:**

```javascript
const jwt = require('jsonwebtoken');
const crypto = require('node:crypto');

class JWTService {
  constructor() {
    // Use strong secrets (at least 32 bytes)
    this.accessTokenSecret = process.env.JWT_ACCESS_SECRET;
    this.refreshTokenSecret = process.env.JWT_REFRESH_SECRET;
    this.accessTokenExpiry = '15m';
    this.refreshTokenExpiry = '7d';
  }
  
  generateTokens(user) {
    const payload = {
      sub: user.id,
      email: user.email,
      role: user.role,
      // Add jti (JWT ID) for revocation tracking
      jti: crypto.randomBytes(16).toString('hex')
    };
    
    const accessToken = jwt.sign(payload, this.accessTokenSecret, {
      expiresIn: this.accessTokenExpiry,
      issuer: process.env.JWT_ISSUER,
      audience: process.env.JWT_AUDIENCE
    });
    
    const refreshToken = jwt.sign(
      { sub: user.id, jti: payload.jti },
      this.refreshTokenSecret,
      { expiresIn: this.refreshTokenExpiry }
    );
    
    return { accessToken, refreshToken };
  }
  
  verifyAccessToken(token) {
    try {
      return jwt.verify(token, this.accessTokenSecret, {
        issuer: process.env.JWT_ISSUER,
        audience: process.env.JWT_AUDIENCE
      });
    } catch (err) {
      if (err instanceof jwt.TokenExpiredError) {
        throw new Error('Token expired');
      }
      throw new Error('Invalid token');
    }
  }
  
  verifyRefreshToken(token) {
    return jwt.verify(token, this.refreshTokenSecret);
  }
}
```

**Authentication Middleware:**

```javascript
function authenticate(req, res, next) {
  const authHeader = req.headers.authorization;
  
  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'No token provided' });
  }
  
  const token = authHeader.substring(7);
  
  try {
    const decoded = jwtService.verifyAccessToken(token);
    req.user = decoded;
    next();
  } catch (err) {
    // Check for specific error types
    if (err.message === 'Token expired') {
      return res.status(401).json({ error: 'Token expired', code: 'TOKEN_EXPIRED' });
    }
    return res.status(401).json({ error: 'Invalid token' });
  }
}
```

### OAuth2 / OpenID Connect

OAuth2 is an **authorization framework** that allows third‑party applications to obtain limited access to user resources without exposing credentials.

**Authorization Code Flow (PKCE) — Most Secure:**

```
┌──────────┐                                          ┌──────────┐
│  Client  │                                          │  Auth    │
│  (SPA)   │                                          │  Server  │
└────┬─────┘                                          └────┬─────┘
     │                                                      │
     │  1. Generate code_verifier & code_challenge         │
     │─────────────────────────────────────────────────────▶│
     │                                                      │
     │  2. Redirect to /authorize with code_challenge      │
     │◀─────────────────────────────────────────────────────│
     │                                                      │
     │  3. User authenticates and consents                 │
     │◀─────────────────────────────────────────────────────│
     │                                                      │
     │  4. Authorization code returned                     │
     │◀─────────────────────────────────────────────────────│
     │                                                      │
     │  5. Exchange code + code_verifier for tokens        │
     │─────────────────────────────────────────────────────▶│
     │                                                      │
     │  6. Access & Refresh tokens returned                │
     │◀─────────────────────────────────────────────────────│
```

**Implementation with `openid-client`:**

```javascript
const { Issuer, generators } = require('openid-client');

class OAuth2Service {
  constructor() {
    this.client = null;
  }
  
  async initialize() {
    const issuer = await Issuer.discover(process.env.OAUTH_ISSUER);
    this.client = new issuer.Client({
      client_id: process.env.OAUTH_CLIENT_ID,
      client_secret: process.env.OAUTH_CLIENT_SECRET,
      redirect_uris: [process.env.OAUTH_REDIRECT_URI],
      response_types: ['code']
    });
  }
  
  getAuthorizationUrl(state) {
    const codeVerifier = generators.codeVerifier();
    const codeChallenge = generators.codeChallenge(codeVerifier);
    
    // Store code_verifier in session for later exchange
    return {
      url: this.client.authorizationUrl({
        scope: 'openid profile email',
        code_challenge: codeChallenge,
        code_challenge_method: 'S256',
        state
      }),
      codeVerifier
    };
  }
  
  async exchangeCode(code, codeVerifier) {
    const tokenSet = await this.client.callback(
      process.env.OAUTH_REDIRECT_URI,
      { code },
      { code_verifier: codeVerifier }
    );
    
    return {
      accessToken: tokenSet.access_token,
      refreshToken: tokenSet.refresh_token,
      idToken: tokenSet.id_token,
      expiresIn: tokenSet.expires_in
    };
  }
  
  async refreshTokens(refreshToken) {
    const tokenSet = await this.client.refresh(refreshToken);
    return {
      accessToken: tokenSet.access_token,
      refreshToken: tokenSet.refresh_token || refreshToken,
      expiresIn: tokenSet.expires_in
    };
  }
}
```

## 3. Token Rotation

Token rotation is a security pattern where **refresh tokens are replaced** with new ones after each use. This detects token replay attacks and limits damage from stolen tokens.

### Refresh Token Rotation Implementation

```javascript
class TokenRotationService {
  constructor() {
    this.usedRefreshTokens = new Set(); // In production: Redis
    this.tokenFamily = new Map();       // Track token lineage
  }
  
  async rotateRefreshToken(userId, oldRefreshToken) {
    // Check if token has been used before (replay attack)
    if (await this.isTokenUsed(oldRefreshToken)) {
      // Token reuse detected! Revoke entire family
      await this.revokeTokenFamily(userId);
      throw new Error('Token reuse detected — family revoked');
    }
    
    // Mark old token as used
    await this.markTokenUsed(oldRefreshToken, userId);
    
    // Generate new token pair
    const newTokens = this.generateTokens(userId);
    
    // Link new refresh token to the family
    const familyId = this.getFamilyId(oldRefreshToken);
    this.linkToFamily(newTokens.refreshToken, familyId);
    
    return newTokens;
  }
  
  async isTokenUsed(token) {
    // Check Redis for used token
    return this.usedRefreshTokens.has(this.hashToken(token));
  }
  
  async markTokenUsed(token, userId) {
    const hashed = this.hashToken(token);
    this.usedRefreshTokens.add(hashed);
    // Store with TTL matching refresh token expiry
    await redis.setex(`used_refresh:${hashed}`, 7 * 24 * 3600, userId);
  }
  
  async revokeTokenFamily(userId) {
    // Revoke all tokens in the family
    await redis.del(`token_family:${userId}`);
    // Invalidate all user sessions (optional)
    await redis.set(`user:${userId}:token_version`, Date.now());
  }
  
  hashToken(token) {
    return crypto.createHash('sha256').update(token).digest('hex');
  }
  
  getFamilyId(token) {
    try {
      const decoded = jwt.decode(token);
      return decoded.familyId;
    } catch {
      return null;
    }
  }
}
```

### Token Rotation Flow

```javascript
// Refresh endpoint
app.post('/auth/refresh', async (req, res) => {
  const { refreshToken } = req.body;
  
  try {
    // Verify refresh token
    const decoded = jwtService.verifyRefreshToken(refreshToken);
    
    // Check if token version matches (invalidation)
    const currentVersion = await redis.get(`user:${decoded.sub}:token_version`);
    if (decoded.version !== currentVersion) {
      return res.status(401).json({ error: 'Token revoked' });
    }
    
    // Rotate tokens
    const newTokens = await tokenRotation.rotateRefreshToken(decoded.sub, refreshToken);
    
    res.json(newTokens);
  } catch (err) {
    if (err.message.includes('reuse')) {
      // Security alert — possible token theft
      console.error('Refresh token reuse detected for user:', decoded?.sub);
      await securityAlerts.notify('TOKEN_REUSE', { userId: decoded?.sub });
      
      // Force re‑authentication
      res.status(401).json({ error: 'Session invalidated', requireReauth: true });
    } else {
      res.status(401).json({ error: 'Invalid refresh token' });
    }
  }
});
```

### Silent Refresh in Frontend

```javascript
// Frontend token manager with rotation awareness
class TokenManager {
  constructor() {
    this.accessToken = null;
    this.refreshToken = null;
    this.refreshPromise = null;
  }
  
  async getAccessToken() {
    if (this.isTokenValid(this.accessToken)) {
      return this.accessToken;
    }
    
    // Deduplicate concurrent refresh requests
    if (!this.refreshPromise) {
      this.refreshPromise = this.refreshTokens();
    }
    
    try {
      const tokens = await this.refreshPromise;
      return tokens.accessToken;
    } finally {
      this.refreshPromise = null;
    }
  }
  
  async refreshTokens() {
    const response = await fetch('/auth/refresh', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ refreshToken: this.refreshToken })
    });
    
    if (!response.ok) {
      const error = await response.json();
      if (error.requireReauth) {
        // Token reuse detected — force login
        this.clearTokens();
        window.location.href = '/login?reason=security';
      }
      throw new Error('Refresh failed');
    }
    
    const tokens = await response.json();
    this.setTokens(tokens.accessToken, tokens.refreshToken);
    return tokens;
  }
}
```

## 4. SQL & NoSQL Injection Prevention

### SQL Injection Prevention (PostgreSQL / MySQL)

**Never concatenate user input into queries.**

```javascript
// ❌ VULNERABLE — DO NOT USE
const query = `SELECT * FROM users WHERE email = '${email}' AND password = '${password}'`;
// Attacker input: email = ' OR '1'='1' -- 
// Results in: SELECT * FROM users WHERE email = '' OR '1'='1' --' AND password = ''

// ✅ SAFE: Parameterized queries
const result = await db.query(
  'SELECT * FROM users WHERE email = $1 AND password = $2',
  [email, password]
);
```

**Dynamic Table/Column/Group BY — Whitelist Validation:**

```javascript
// When you must have dynamic identifiers (sort column, table name)
const ALLOWED_COLUMNS = new Set(['id', 'name', 'email', 'created_at']);
const ALLOWED_DIRECTIONS = new Set(['ASC', 'DESC']);

function buildOrderByClause(sortBy, sortDir) {
  if (!ALLOWED_COLUMNS.has(sortBy)) {
    throw new Error('Invalid column');
  }
  if (!ALLOWED_DIRECTIONS.has(sortDir.toUpperCase())) {
    throw new Error('Invalid direction');
  }
  return `ORDER BY ${sortBy} ${sortDir.toUpperCase()}`;
}
```

**Using an ORM (Prisma / Sequelize / TypeORM):**

```javascript
// Prisma automatically parameterizes
const users = await prisma.user.findMany({
  where: {
    email: email,
    active: true
  },
  orderBy: {
    createdAt: 'desc'
  }
});
```

### NoSQL Injection Prevention (MongoDB)

MongoDB queries can be vulnerable when user input is used directly as query operators.

```javascript
// ❌ VULNERABLE
const user = await db.collection('users').findOne({
  username: req.body.username,
  password: req.body.password
});
// Attacker: { "username": "admin", "password": { "$ne": "" } }
// Bypasses password check!

// ✅ SAFE: Validate and sanitize
const { username, password } = req.body;

// Ensure password is a string
if (typeof password !== 'string') {
  throw new Error('Invalid password');
}

const user = await db.collection('users').findOne({
  username: String(username),
  password: hashPassword(password)
});
```

**Sanitization Middleware for MongoDB:**

```javascript
function sanitizeMongoQuery(obj) {
  if (typeof obj !== 'object' || obj === null) return obj;
  
  const sanitized = Array.isArray(obj) ? [] : {};
  
  for (const [key, value] of Object.entries(obj)) {
    // Block MongoDB operators at root level
    if (key.startsWith('$')) {
      continue; // Skip dangerous operators
    }
    
    sanitized[key] = sanitizeMongoQuery(value);
  }
  
  return sanitized;
}

// Usage in route
app.post('/api/search', (req, res) => {
  const sanitizedFilter = sanitizeMongoQuery(req.body.filter);
  const results = await db.collection('items').find(sanitizedFilter).toArray();
});
```

## 5. DDoS Protection

DDoS (Distributed Denial of Service) attacks aim to overwhelm your server with traffic. Protection requires a **layered approach**.

### Application‑Level Rate Limiting

```javascript
const rateLimit = require('express-rate-limit');
const RedisStore = require('rate-limit-redis');
const Redis = require('ioredis');

const redis = new Redis(process.env.REDIS_URL);

// Global limiter
const globalLimiter = rateLimit({
  store: new RedisStore({
    client: redis,
    prefix: 'rl:global:'
  }),
  windowMs: 60 * 1000,      // 1 minute
  max: 100,                 // 100 requests per minute
  standardHeaders: true,
  legacyHeaders: false,
  message: 'Too many requests, please try again later.',
  skipSuccessfulRequests: false
});

// Strict limiter for auth endpoints
const authLimiter = rateLimit({
  store: new RedisStore({
    client: redis,
    prefix: 'rl:auth:'
  }),
  windowMs: 15 * 60 * 1000,  // 15 minutes
  max: 5,                    // 5 failed attempts
  skipSuccessfulRequests: true, // Only count failures
  keyGenerator: (req) => req.body.email || req.ip
});

app.use('/api/', globalLimiter);
app.use('/auth/login', authLimiter);
```

### Socket / Connection Limiting

```javascript
const http = require('node:http');
const server = http.createServer(app);

// Limit concurrent connections
server.maxConnections = 10000;

// Track connections per IP
const connectionsPerIP = new Map();

server.on('connection', (socket) => {
  const ip = socket.remoteAddress;
  const count = (connectionsPerIP.get(ip) || 0) + 1;
  connectionsPerIP.set(ip, count);
  
  if (count > 50) {
    socket.destroy();
    return;
  }
  
  socket.once('close', () => {
    const current = connectionsPerIP.get(ip);
    if (current === 1) {
      connectionsPerIP.delete(ip);
    } else {
      connectionsPerIP.set(ip, current - 1);
    }
  });
});

// Timeout slow connections
server.timeout = 10000; // 10 seconds
server.keepAliveTimeout = 5000;
server.headersTimeout = 6000;
```

### Header‑by‑Header Explanation

| Header | Purpose | Recommended Value |
|--------|---------|-------------------|
| **Content‑Security‑Policy** | Prevents XSS and data injection | Strict, allow only trusted sources |
| **Strict‑Transport‑Security** | Forces HTTPS | `max-age=31536000; includeSubDomains; preload` |
| **X‑Frame‑Options** | Prevents clickjacking | `DENY` or `SAMEORIGIN` |
| **X‑Content‑Type‑Options** | Prevents MIME sniffing | `nosniff` |
| **Referrer‑Policy** | Controls referrer information | `strict-origin-when-cross-origin` |
| **Cross‑Origin‑Opener‑Policy** | Isolates browsing context | `same-origin` |
| **Cross‑Origin‑Embedder‑Policy** | Controls resource loading | `require-corp` (for SharedArrayBuffer) |

## 7. Rate Limiting (Advanced)

Beyond basic rate limiting, enterprise applications need sophisticated throttling strategies.

### Token Bucket Algorithm (Redis‑Based)

```javascript
class TokenBucketRateLimiter {
  constructor(redis, options = {}) {
    this.redis = redis;
    this.capacity = options.capacity || 100;      // Max tokens
    this.refillRate = options.refillRate || 10;   // Tokens per second
    this.refillInterval = 1000;                   // 1 second
  }
  
  async consume(key, tokens = 1) {
    const bucketKey = `bucket:${key}`;
    
    // Lua script for atomic token bucket
    const script = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local refillRate = tonumber(ARGV[2])
      local now = tonumber(ARGV[3])
      local requested = tonumber(ARGV[4])
      
      local bucket = redis.call('HMGET', key, 'tokens', 'lastRefill')
      local tokens = tonumber(bucket[1]) or capacity
      local lastRefill = tonumber(bucket[2]) or now
      
      local elapsed = math.max(0, now - lastRefill)
      local refill = math.floor(elapsed * refillRate / 1000)
      
      tokens = math.min(capacity, tokens + refill)
      
      local allowed = 0
      if tokens >= requested then
        tokens = tokens - requested
        allowed = 1
      end
      
      redis.call('HMSET', key, 'tokens', tokens, 'lastRefill', now)
      redis.call('EXPIRE', key, math.ceil(capacity / refillRate) + 10)
      
      return {allowed, tokens}
    `;
    
    const result = await this.redis.eval(
      script, 1, bucketKey,
      this.capacity, this.refillRate, Date.now(), tokens
    );
    
    return {
      allowed: result[0] === 1,
      remainingTokens: result[1]
    };
  }
}

// Usage
const limiter = new TokenBucketRateLimiter(redis, {
  capacity: 100,
  refillRate: 10
});

app.use(async (req, res, next) => {
  const key = req.user?.id || req.ip;
  const result = await limiter.consume(key);
  
  res.setHeader('X-RateLimit-Remaining', result.remainingTokens);
  
  if (!result.allowed) {
    return res.status(429).json({ 
      error: 'Rate limit exceeded',
      retryAfter: Math.ceil((result.remainingTokens * -1) / 10)
    });
  }
  next();
});
```

```javascript
class SlidingWindowRateLimiter {
  async isAllowed(key, windowMs, maxRequests) {
    const now = Date.now();
    const windowStart = now - windowMs;
    
    // Use Redis sorted set with timestamps as scores
    const script = `
      local key = KEYS[1]
      local now = tonumber(ARGV[1])
      local windowStart = tonumber(ARGV[2])
      local maxRequests = tonumber(ARGV[3])
      
      redis.call('ZREMRANGEBYSCORE', key, 0, windowStart)
      
      local count = redis.call('ZCARD', key)
      
      if count < maxRequests then
        redis.call('ZADD', key, now, now)
        redis.call('EXPIRE', key, ARGV[4])
        return 1
      end
      
      return 0
    `;
    
    const result = await this.redis.eval(
      script, 1, key,
      now, windowStart, maxRequests, Math.ceil(windowMs / 1000) + 1
    );
    
    return result === 1;
  }
```

### Hierarchical Rate Limiting (User + IP + Global)

```javascript
class MultiLevelRateLimiter {
  constructor() {
    this.limiters = {
      global: new TokenBucketRateLimiter(redis, { capacity: 10000, refillRate: 1000 }),
      ip: new TokenBucketRateLimiter(redis, { capacity: 500, refillRate: 50 }),
      user: new TokenBucketRateLimiter(redis, { capacity: 1000, refillRate: 100 }),
      endpoint: new TokenBucketRateLimiter(redis, { capacity: 100, refillRate: 10 })
    };
  }
  
  async check(req) {
    const checks = [];
    
    // Global check
    checks.push(this.limiters.global.consume('global'));
    
    // IP‑based check
    checks.push(this.limiters.ip.consume(`ip:${req.ip}`));
    
    // User‑based check (if authenticated)
    if (req.user) {
      checks.push(this.limiters.user.consume(`user:${req.user.id}`));
    }
    
    // Endpoint‑specific check
    const endpoint = `${req.method}:${req.path}`;
    checks.push(this.limiters.endpoint.consume(`endpoint:${endpoint}`));
    
    const results = await Promise.all(checks);
    
    // If any limiter rejects, block the request
    return results.every(r => r.allowed);
  }
}
```

### Secrets Management

```javascript
// NEVER hardcode secrets
// Use environment variables or secret managers (Vault, AWS Secrets Manager)

const { SecretsManagerClient, GetSecretValueCommand } = require('@aws-sdk/client-secrets-manager');

class SecretService {
  constructor() {
    this.client = new SecretsManagerClient({ region: process.env.AWS_REGION });
    this.cache = new Map();
  }
  
  async getSecret(secretName) {
    if (this.cache.has(secretName)) {
      return this.cache.get(secretName);
    }
    
    const command = new GetSecretValueCommand({ SecretId: secretName });
    const response = await this.client.send(command);
    
    const secret = JSON.parse(response.SecretString);
    this.cache.set(secretName, secret);
    
    // Auto‑refresh cache before expiry
    setTimeout(() => this.cache.delete(secretName), 5 * 60 * 1000);
    
    return secret;
  }
}
```

## Summary: Security Checklist for Production

| Category | Must Implement |
|----------|----------------|
| **Authentication** | JWT with short expiry, refresh token rotation, secure httpOnly cookies |
| **Authorization** | RBAC or ABAC middleware, tenant isolation |
| **Input Validation** | Schema validation (Zod/Joi), sanitization, parameterized queries |
| **Headers** | Helmet with strict CSP, HSTS, no MIME sniffing |
| **Rate Limiting** | Multi‑level (global, IP, user, endpoint), Redis‑based |
| **DDoS Protection** | Connection limits, timeout configuration, cloud WAF |
| **Secrets** | Environment variables, secret manager, never in code |
| **Logging** | Structured logging, redact sensitive data, audit trails |
| **Monitoring** | Anomaly detection, failed auth alerts, rate limit violations |

**Key Takeaways:**
1. **Defense in depth** — No single security measure is sufficient
2. **Never trust user input** — Validate, sanitize, parameterize
3. **Rotate tokens** — Refresh token rotation prevents replay attacks
4. **Set security headers** — Helmet provides a solid baseline
5. **Rate limit everywhere** — Protect auth endpoints especially
6. **Monitor and alert** — Detect attacks early
7. **Keep dependencies updated** — Run `npm audit` regularly
